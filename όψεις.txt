CREATE VIEW Καρτέλα_αθλητή AS
SELECT επώνυμο, όνομα, τηλέφωνο, διεύθυνση, ηλικία 
FROM ΑΘΛΗΤΗΣ 
ORDER BY επώνυμο, όνομα;

CREATE VIEW Καρτέλα_προπονητή AS 
SELECT επώνυμο, όνομα, τηλέφωνο, διεύθυνση, ειδικότητα 
FROM ΠΡΟΠΟΝΗΤΗΣ
ORDER BY επώνυμο, όνομα;

CREATE VIEW Τμήματα_ομάδας (κωδικός, όνομα, επώνυμο_προπονητή, όνομα_προποητή, άθλημα, μέλη_τμήματος) AS 
SELECT ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός, ΤΜΗΜΑ_ΟΜΑΔΑΣ.όνομα, ΠΡΟΠΟΝΗΤΗΣ.επώνυμο, ΠΡΟΠΟΝΗΤΗΣ.όνομα, ΑΘΛΗΜΑ.όνομα, COUNT(*)
FROM ΤΜΗΜΑ_ΟΜΑΔΑΣ, ΠΡΟΠΟΝΗΤΗΣ, ΑΘΛΗΤΗΣ, ΑΘΛΗΜΑ, είναι_μέλος
WHERE είναι_μέλος.FK2_κωδικός = ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός
AND είναι_μέλος.FK1_κωδικός = ΑΘΛΗΤΗΣ.κωδικός
AND ΤΜΗΜΑ_ΟΜΑΔΑΣ.FK2_κωδικός = ΠΡΟΠΟΝΗΤΗΣ.κωδικός
GROUP BY ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός, ΠΡΟΠΟΝΗΤΗΣ.επώνυμο, ΠΡΟΠΟΝΗΤΗΣ.όνομα, ΑΘΛΗΜΑ.όνομα;
