1)
SELECT κωδικός, (επώνυμο || ' ' || όνομα) Ονοματεπώνυμο
FROM ΠΡΟΠΟΝΗΤΗΣ
ORDER BY Ονοματεπώνυμο ASC;

2)
SELECT κωδικός, (επώνυμο || ' ' || όνομα) Ονοματεπώνυμο
FROM ΑΘΛΗΤΗΣ
ORDER BY Ονοματεπώνυμο ASC;

3)
SELECT  COUNT(*)
FROM ΠΡΟΠΟΝΗΤΗΣ;

4)
SELECT  COUNT(*)
FROM ΑΘΛΗΤΗΣ;

5)
SELECT όνομα
FROM ΑΘΛΗΜΑ
ORDER BY κωδικός ASC;

6)
SELECT  COUNT(*)
FROM ΑΘΛΗΜΑ;

7)
SELECT (επώνυμο || ' ' || όνομα) Ονοματεπώνυμο
FROM ΑΘΛΗΤΗΣ, είναι_μέλος
WHERE ΑΘΛΗΤΗΣ.κωδικός = είναι_μέλος.FK1_κωδικός AND είναι_μέλος.FK2_κωδικός = 3  
ORDER BY Ονοματεπώνυμο ASC;

8)
SELECT (ΠΡΟΠΟΝΗΤΗΣ.επώνυμο || ' ' || ΠΡΟΠΟΝΗΤΗΣ.όνομα) Ονοματεπώνυμο
FROM ΠΡΟΠΟΝΗΤΗΣ, διδάσκει, ΑΘΛΗΜΑ
WHERE ΑΘΛΗΜΑ.κωδικός = διδάσκει.FK1_κωδικός AND ΠΡΟΠΟΝΗΤΗΣ.κωδικός = διδάσκει.FK2_κωδικός AND ΑΘΛΗΜΑ.όνομα = 'Μπάσκετ';

9)
SELECT (επώνυμο || ' ' || όνομα) Ονοματεπώνυμο
FROM ΑΘΛΗΤΗΣ
WHERE κωδικός IN (SELECT FK1_κωδικός
		  FROM μαθαίνει
		  GROUP BY FK1_κωδικός
		  HAVING COUNT(*) >= 3);

10)
SELECT DISTINCT ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός
FROM ΤΜΗΜΑ_ΟΜΑΔΑΣ, ΠΡΟΠΟΝΗΤΗΣ
WHERE ΠΡΟΠΟΝΗΤΗΣ.κωδικός = ΤΜΗΜΑ_ΟΜΑΔΑΣ.FK1_κωδικός AND ΠΡΟΠΟΝΗΤΗΣ.επώνυμο = 'Ντζουμάνης' AND ΠΡΟΠΟΝΗΤΗΣ.όνομα = 'Αθανάσιος';

11)
SELECT ΑΘΛΗΜΑ.όνομα
FROM ΑΘΛΗΜΑ, ΑΘΛΗΤΗΣ, μαθαίνει
WHERE μαθαίνει.FK1_κωδικός = ΑΘΛΗΤΗΣ.κωδικός
AND μαθαίνει.FK2_κωδικός = ΑΘΛΗΜΑ.κωδικός
AND (ΑΘΛΗΤΗΣ.επώνυμο || ' ' || ΑΘΛΗΤΗΣ.όνομα) = 'Στάμος Κων/νος'
ORDER BY ΑΘΛΗΜΑ.όνομα ASC;

12)
SELECT COUNT(*)
FROM ΠΡΟΠΟΝΗΤΗΣ, ΤΜΗΜΑ_ΟΜΑΔΑΣ
WHERE ΠΡΟΠΟΝΗΤΗΣ.κωδικός = ΤΜΗΜΑ_ΟΜΑΔΑΣ.FK1_κωδικός
AND (ΠΡΟΠΟΝΗΤΗΣ.επώνυμο || ' ' || ΠΡΟΠΟΝΗΤΗΣ.όνομα) = 'Παπούλιας Παναγιώτης';

13)
SELECT (ΠΡΟΠΟΝΗΤΗΣ.επώνυμο || ' ' || ΠΡΟΠΟΝΗΤΗΣ.όνομα) Ονοματεπώνυμο, COUNT(*) Αρ_τμημάτων
FROM ΠΡΟΠΟΝΗΤΗΣ, ΤΜΗΜΑ_ΟΜΑΔΑΣ
WHERE ΠΡΟΠΟΝΗΤΗΣ.κωδικός = ΤΜΗΜΑ_ΟΜΑΔΑΣ.FK1_κωδικός
GROUP BY Ονοματεπώνυμο
ORDER BY COUNT(*) ASC;

14)
SELECT ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός 
FROM ΤΜΗΜΑ_ΟΜΑΔΑΣ
WHERE ΤΜΗΜΑ_ΟΜΑΔΑΣ.όνομα = 'μπάσκετ_π' OR ΤΜΗΜΑ_ΟΜΑΔΑΣ.όνομα = 'μπάσκετ_γ' OR ΤΜΗΜΑ_ΟΜΑΔΑΣ.όνομα = 'μπάσκετ_α'
ORDER BY ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός ASC;

15)
SELECT (ΑΘΛΗΤΗΣ.επώνυμο || ' ' || ΑΘΛΗΤΗΣ.όνομα) Αθλητής, ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός Τμήμα, ΑΘΛΗΜΑ.όνομα Άθλημα
FROM ΑΘΛΗΤΗΣ, ΤΜΗΜΑ_ΟΜΑΔΑΣ, ΑΘΛΗΜΑ, είναι_μέλος, μαθαίνει
WHERE είναι_μέλος.FK1_κωδικός = ΑΘΛΗΤΗΣ.κωδικός
AND είναι_μέλος.FK2_κωδικός = ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός
AND μαθαίνει.FK1_κωδικός = ΑΘΛΗΤΗΣ.κωδικός
AND μαθαίνει.FK2_κωδικός = ΑΘΛΗΜΑ.κωδικός
ORDER BY Αθλητής;

16)
SELECT (ΠΡΟΠΟΝΗΤΗΣ.επώνυμο || ' ' || ΠΡΟΠΟΝΗΤΗΣ.όνομα) Προπονητής, ΤΜΗΜΑ_ΟΜΑΔΑΣ.κωδικός Τμήμα, ΑΘΛΗΜΑ.όνομα Άθλημα
FROM ΠΡΟΠΟΝΗΤΗΣ, ΤΜΗΜΑ_ΟΜΑΔΑΣ, ΑΘΛΗΜΑ, διδάσκει
WHERE ΠΡΟΠΟΝΗΤΗΣ.κωδικός = ΤΜΗΜΑ_ΟΜΑΔΑΣ.FK1_κωδικός
AND διδάσκει.FK1_κωδικός = ΑΘΛΗΜΑ.κωδικός
AND διδάσκει.FK2_κωδικός = ΠΡΟΠΟΝΗΤΗΣ.κωδικός
ORDER BY Προπονητής;


17)
SELECT *
FROM ΕΠΑΓΓΕΛΜΑΤΙΚΟ
ORDER BY κωδικός ASC;
